接口调用示例

# SM3 摘要
curl -X POST http://localhost:9000/Digest \
  -H "Content-Type: application/json" \
  -d '{"inData":"aGVsbG8="}'

# SM2 签名（keyIndex=1，需配置 mock_certs.toml）
curl -X POST http://localhost:9000/SignData \
  -H "Content-Type: application/json" \
  -d '{"keyIndex":1,"password":"12345678","inData":"aGVsbG8="}'

# 验证签名
curl -X POST http://localhost:9000/VerifySignedData \
  -H "Content-Type: application/json" \
  -d '{"certBase64":"...","inData":"aGVsbG8=","signData":"..."}'

# 数字信封加密
curl -X POST http://localhost:9000/envelopeEnc \
  -H "Content-Type: application/json" \
  -d '{"certBase64":"...","inData":"aGVsbG8="}'
import requests, base64, json

SVS_URL = "http://localhost:9000"

# SM3 摘要
resp = requests.post(f"{SVS_URL}/Digest", json={
    "inData": base64.b64encode(b"hello").decode()
})
print("digest =", resp.json()["hashData"])

# SM2 签名
resp = requests.post(f"{SVS_URL}/SignData", json={
    "keyIndex": 1,
    "password": "12345678",
    "inData": base64.b64encode(b"hello").decode()
})
sig = resp.json()
print("signData =", sig["signData"])

# 验证签名
resp = requests.post(f"{SVS_URL}/VerifySignedData", json={
    "certBase64": sig["certBase64"],
    "inData": base64.b64encode(b"hello").decode(),
    "signData": sig["signData"]
})
print("verify =", resp.json()["respValue"])  # 0 = 成功
